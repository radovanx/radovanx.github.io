Token Fundamentals — BPE, Roles, and Context Windows

Before optimizing tokens, understand what they are. Tokens are subword units — not words, characters, or bytes. Most modern LLMs use Byte Pair Encoding (BPE): start with individual characters, iteratively merge the most common adjacent pairs, stop at ~100,000 tokens.

Message Roles and Cost Asymmetry

Message roleToken behavior
System promptSent on every request; cached after first use
User messageInput tokens; cheaper rate
Assistant responseOutput tokens; 3–5× more expensive than input
Reasoning tokensHidden from response but billed; add latency
Tool callsInput + output; tool schemas add overhead

Key insight: Output tokens cost 3–5× more than input tokens. Compressing responses (Caveman) delivers more ROI than compressing prompts — optimize output first, then input, then context overhead.

const enc = encoding_for_model("gpt-4");

function countTokens(text: string): number {
  return enc.encode(text).length;
  // "Hello, world!" → 4 tokens
  // "function tokenize(text: string): number[]" → 10 tokens
}

function estimateCost(inputTokens: number, outputTokens: number) {
  const inputRate = 3.0 / 1_000_000;   // $3/M input tokens (Claude Sonnet 4.6)
  const outputRate = 15.0 / 1_000_000; // $15/M output tokens (5× input)
  const total = inputTokens * inputRate + outputTokens * outputRate;
  const outputShare = ((outputTokens * outputRate / total) * 100).toFixed(0);
  return { total: `$${total.toFixed(4)}`, outputShare: `${outputShare}% from output` };
}
// estimateCost(1000, 500) → { outputShare: "71% from output" }


<?php
function estimateTokenCount(string $text): int {
    // Rule of thumb: ~0.75 tokens/word + punctuation overhead
    $words = preg_split('/\s+/', trim($text), -1, PREG_SPLIT_NO_EMPTY);
    return (int) ceil(count($words) * 1.3);
}

function calculateCostShare(int $inputTokens, int $outputTokens): string {
    $inputCost  = $inputTokens  * (3.00  / 1_000_000);
    $outputCost = $outputTokens * (15.00 / 1_000_000);
    $share = $outputCost / ($inputCost + $outputCost) * 100;
    return sprintf('%.0f%% of cost from output tokens', $share);
}
// calculateCostShare(1000, 500) → "71% of cost from output tokens"
